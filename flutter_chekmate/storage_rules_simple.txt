rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Profile images - 5MB max
    match /profile_images/{userId}/{file} {
      allow read: if request.auth != null;
      allow write: if request.auth != null 
                   && request.auth.uid == userId 
                   && request.resource.contentType.matches('image/.*')
                   && request.resource.size < 5242880;
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // Legacy users path - 5MB max
    match /users/{userId}/{file} {
      allow read: if request.auth != null;
      allow write: if request.auth != null 
                   && request.auth.uid == userId 
                   && request.resource.contentType.matches('image/.*')
                   && request.resource.size < 5242880;
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
    
    // Post images - 50MB max
    match /post_images/{postId}/{file} {
      allow read: if request.auth != null;
      allow write: if request.auth != null 
                   && request.resource.contentType.matches('image/.*')
                   && request.resource.size < 52428800;
      allow delete: if request.auth != null;
    }
    
    // Post videos - 50MB max
    match /post_videos/{postId}/{file} {
      allow read: if request.auth != null;
      allow write: if request.auth != null 
                   && request.resource.contentType.matches('video/.*')
                   && request.resource.size < 52428800;
      allow delete: if request.auth != null;
    }
    
    // Legacy posts path - 50MB max
    match /posts/{postId}/{file} {
      allow read: if request.auth != null;
      allow write: if request.auth != null 
                   && (request.resource.contentType.matches('image/.*') 
                       || request.resource.contentType.matches('video/.*'))
                   && request.resource.size < 52428800;
      allow delete: if request.auth != null;
    }
    
    // Story images - 25MB max
    match /story_images/{storyId}/{file} {
      allow read: if request.auth != null;
      allow write: if request.auth != null 
                   && request.resource.contentType.matches('image/.*')
                   && request.resource.size < 26214400;
      allow delete: if request.auth != null;
    }
    
    // Story videos - 25MB max
    match /story_videos/{storyId}/{file} {
      allow read: if request.auth != null;
      allow write: if request.auth != null 
                   && request.resource.contentType.matches('video/.*')
                   && request.resource.size < 26214400;
      allow delete: if request.auth != null;
    }
    
    // Voice messages - 5MB max
    match /voice_messages/{userId}/{messageId}/{file} {
      allow read: if request.auth != null;
      allow write: if request.auth != null 
                   && request.auth.uid == userId 
                   && request.resource.contentType.matches('audio/.*')
                   && request.resource.size < 5242880;
      allow delete: if request.auth != null && request.auth.uid == userId;
    }
  }
}
